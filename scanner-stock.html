async function buscarStock(codigoForzado = null) { const input = document.getElementById('codigoInput'); const codigo = codigoForzado || input.value.trim(); if (!codigo) return; cerrarEscanner(); limpiarChecks(); mostrarVista('loader'); try { const response = await fetch(API_URL + codigo); const data = await response.json(); if (data.error || !data.nombre || data.nombre === "No detectado") throw new Error(); currentProduct = data; // --- NUEVA LÓGICA DE DETECCIÓN AGRESIVA PARA DESCONTINUADOS --- // Convertimos todo el objeto a texto y buscamos la palabra "DESCONTINUADO" const rawDataString = JSON.stringify(data).toUpperCase(); const esDescontinuado = rawDataString.includes("DESCONTINUADO"); const banner = document.getElementById('resBanner'); const resCodigo = document.getElementById('resCodigo'); const resNombre = document.getElementById('resNombre'); if (esDescontinuado) { // Cambiar a ROJO si es descontinuado banner.classList.remove('bg-blue-600'); banner.classList.add('bg-red-600'); resCodigo.innerHTML = ` PRODUCTO DESCONTINUADO `; resNombre.classList.add('text-red-700'); // Nombre en rojo también } else { // Azul normal si está vigente banner.classList.remove('bg-red-600'); banner.classList.add('bg-blue-600'); resCodigo.innerText = "CÓDIGO: " + data.codigo; resNombre.classList.remove('text-red-700'); } // ------------------------------------------------------------- document.getElementById('resNombre').innerText = data.nombre; document.getElementById('resPrecio').innerText = data.precio ? "$" + data.precio.toLocaleString('es-CL') : "---"; document.getElementById('resStock').innerText = data.stock ?? 0; // Color del stock (Rojo si es 0 o menos) const sBox = document.getElementById('stockBox'); if (parseInt(data.stock) <= 0) { sBox.className = "p-4 rounded-2xl border border-red-100 bg-red-50 text-red-600 text-center flex flex-col items-center justify-center"; document.getElementById('stockLabel').innerText = "SIN STOCK / NEGATIVO"; } else { sBox.className = "p-4 rounded-2xl border border-green-50 bg-green-50 text-green-600 text-center flex flex-col items-center justify-center"; document.getElementById('stockLabel').innerText = "EN STOCK"; } mostrarVista('resultado'); } catch (error) { mostrarVista('errorMsg'); } }
